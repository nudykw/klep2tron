### 1. **Структура данных для Manual режима**

```rust
// Предложение для хранения ручных правок
#[derive(Clone, Debug, Serialize, Deserialize)]
pub struct ManualSlicingOverrides {
    /// Маппинг: индекс треугольника в оригинальном меше -> целевая часть
    pub triangle_assignments: HashMap<usize, ActorPart>,
    
    /// Флаг: были ли применены ручные правки
    pub has_overrides: bool,
    
    /// Версия для совместимости при загрузке старых проектов
    pub version: u32,
}

#[derive(Clone, Copy, Debug, Serialize, Deserialize, PartialEq, Eq)]
pub enum ActorPart {
    Head,
    Body,
    Legs, // или Engine, в зависимости от терминологии
}
```

### 2. **Оптимизация Lasso Selection**

```rust
// Псевдокод для эффективного Lasso Selection
fn select_triangles_in_lasso(
    mesh: &Mesh,
    lasso_points: &[Vec2], // экранные координаты
    camera: &Camera,
    view_matrix: &Mat4,
) -> Vec<usize> {
    // 1. Построить AABB лассо в экранном пространстве
    let lasso_aabb = compute_aabb_2d(lasso_points);
    
    // 2. Быстрая фильтрация: проверить только треугольники,
    //    чьи проекции пересекаются с AABB
    let candidates = mesh.triangles.iter()
        .enumerate()
        .filter(|(_, tri)| {
            let projected = project_triangle(tri, view_matrix, camera);
            aabb_intersects(&lasso_aabb, &projected)
        });
    
    // 3. Точная проверка: point-in-polygon для каждой вершины
    candidates
        .filter(|(_, tri)| {
            tri.vertices.iter().any(|v| {
                let screen_pos = project_vertex(v, view_matrix, camera);
                point_in_polygon(&screen_pos, lasso_points)
            })
        })
        .map(|(idx, _)| idx)
        .collect()
}
```

### 3. **Интеграция с Undo/Redo**

```rust
// Команда для системы Undo/Redo
#[derive(Clone)]
pub struct ReassignTrianglesCommand {
    pub triangle_indices: Vec<usize>,
    pub from_part: ActorPart,
    pub to_part: ActorPart,
}

impl UndoableCommand for ReassignTrianglesCommand {
    fn execute(&self, state: &mut EditorState) {
        // Перенести треугольники
        for &idx in &self.triangle_indices {
            state.manual_overrides.triangle_assignments
                .insert(idx, self.to_part);
        }
        state.rebuild_part_meshes();
    }
    
    fn undo(&self, state: &mut EditorState) {
        // Вернуть треугольники обратно
        for &idx in &self.triangle_indices {
            state.manual_overrides.triangle_assignments
                .insert(idx, self.from_part);
        }
        state.rebuild_part_meshes();
    }
}
```
